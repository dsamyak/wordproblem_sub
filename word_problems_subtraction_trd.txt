1. TECHNICAL OVERVIEW

This document specifies the architecture, component design, state management, data
models, simulation logic, gamification implementation, audio pipeline, and quality
standards for the "Word Problems Using Subtraction" interactive lesson module
(Lesson 3.4) within Intellia SG's Grade 1 Math program.

The module is a React 18 application (Vite + JSX), structured identically to the
reference repository https://github.com/dsamyak/numberbound, and styled to exactly
match https://numberbound.vercel.app/. It will be embedded in WordPress at:

https://intelliasg.com/courses/grade-1-math/lessons/simple-word-problems/

The audio pipeline uses ElevenLabs exclusively (Voice: Alice, Voice ID:
Xb7hH8MSUJpSbSDYk0k2, Model: eleven_multilingual_v2), following the pre-generation
+ dynamic fallback architecture defined in audio_generation_pipeline.md (the Word
Problems Using Addition reference module). No Web Speech API fallback is used.

  const xpGained = calcXP(state.attemptCount + 1, state.hintsUsed, state.streak);

  const newWorldScores = [...state.worldScores];

    worldScores: newWorldScores,

    attemptCount: 0,
  };
}

  return {
    ...state,
    streak: 0,
    attemptCount: state.attemptCount + 1,
  };
}

8. RANDOMISATION ENGINE

8.1 Fisher-Yates Shuffle

// utils/shuffle.js
export function shuffleArray(array) {
  const arr = [...array];
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr;
}

export function generateSessionQuestions(bank) {
  // Group by type, take 10 from each (shuffled within type), then global shuffle
  const byType = bank.reduce((acc, q) => {
    (acc[q.type] = acc[q.type] || []).push(q);
    return acc;
  }, {});
  const selected = Object.values(byType).flatMap(qs =>
    shuffleArray(qs).slice(0, 10)
  );
  return shuffleArray(selected); // Final global shuffle of all 100
}

8.2 MCQ Distractor Generation

// utils/scoring.js
export function generateDistractors(correct, min = 0, max = 20, count = 3) {
  const distractors = new Set();
  let attempts = 0;
  while (distractors.size < count && attempts < 50) {
    const offset = (Math.random() > 0.5 ? 1 : -1) * (Math.ceil(Math.random() * 3));
    const d = correct + offset;
    if (d >= min && d <= max && d !== correct) distractors.add(d);
    attempts++;
  }
  // Ensure fallback distractors
  for (const d of [correct - 1, correct + 1, correct + 2, correct - 2]) {
    if (d >= min && d <= max && d !== correct && distractors.size < count)
      distractors.add(d);
  }
  return shuffleArray([correct, ...[...distractors].slice(0, count)]);
}

8.3 Session Persistence (localStorage)

const SESSION_KEY = 'intellia_wp_sub_v1';
const SESSION_TTL = 86400000; // 24 hours

export function saveSession(state) {
  localStorage.setItem(SESSION_KEY, JSON.stringify({
    phase: state.phase,
    storyPanel: state.storyPanel,
    currentSimStation: state.currentSimStation,
    simStationsComplete: state.simStationsComplete,
    currentQuestion: state.currentQuestion,
    currentWorld: state.currentWorld,
    worldScores: state.worldScores,
    xp: state.xp,
    totalStars: state.totalStars,
    badges: state.badges,
    phaseComplete: state.phaseComplete,
    maxStreak: state.maxStreak,
    audioEnabled: state.audioEnabled,
    timestamp: Date.now(),
  }));
}

export function loadSession() {
  try {
    const raw = localStorage.getItem(SESSION_KEY);
    if (!raw) return null;
    const saved = JSON.parse(raw);
    if (Date.now() - saved.timestamp > SESSION_TTL) {
      localStorage.removeItem(SESSION_KEY);
      return null;
    }
    return saved;
  } catch {
    return null;
  }
}

9.1 XP Calculation

// utils/scoring.js
export function calcXP(attemptNumber, hintsUsed, streak) {
  let base;
  if (attemptNumber === 1 && hintsUsed === 0) base = 10;
  else if (hintsUsed > 0) base = 5;
  else base = 7;
  const streakBonus = streak >= 5 ? 5 : 0;
  return base + streakBonus;
}

  if (correct >= 7) return 2;
  if (correct >= 5) return 1;
  return 0;
}

11.4 Frontend Audio Engine (src/utils/audio.js)

// Matches addition module audio.js exactly, adapted for subtraction module

const elevenLabsCache = new Map(); // In-memory URL cache

export async function getAudioUrl(text, style = 'statement') {
  // 1. Check pre-generated static map
  if (audioMap[text]) return audioMap[text];

  // 2. Check in-memory dynamic cache
  const cacheKey = `${text}::${style}`;
  if (elevenLabsCache.has(cacheKey)) return elevenLabsCache.get(cacheKey);

  // 3. Dynamic generation via ElevenLabs API
  const apiKey = import.meta.env.VITE_ELEVENLABS_API_KEY;
  if (!apiKey) return null; // Silent skip — no fallback

  const VOICE_ID = 'Xb7hH8MSUJpSbSDYk0k2';
  const styleSettings = SPEECH_STYLES[style] || SPEECH_STYLES.statement;

  try {
    const response = await fetch(
      `https://api.elevenlabs.io/v1/text-to-speech/${VOICE_ID}`,
      {
        method: 'POST',
        headers: {
          'xi-api-key': apiKey,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({
          text,
          model_id: 'eleven_multilingual_v2',
          voice_settings: styleSettings,
        }),
      }
    );
    if (!response.ok) return null;
    const blob = await response.blob();
    const url = URL.createObjectURL(blob);
    elevenLabsCache.set(cacheKey, url);
    return url;
  } catch {
    return null; // Silent skip on error
  }
}

const SPEECH_STYLES = {
  statement:    { stability: 0.75, similarity_boost: 0.80, style: 0.10 },
  question:     { stability: 0.60, similarity_boost: 0.75, style: 0.25 },
  encouragement:{ stability: 0.50, similarity_boost: 0.85, style: 0.45 },
  emphasis:     { stability: 0.80, similarity_boost: 0.85, style: 0.15 },
  thinking:     { stability: 0.70, similarity_boost: 0.75, style: 0.20 },
  celebration:  { stability: 0.40, similarity_boost: 0.90, style: 0.55 },
};

// Seamless narration with eager preloading
export async function narrate(segments, audioEnabled) {
  if (!audioEnabled) return;
  for (let i = 0; i < segments.length; i++) {
    const { text, style } = segments[i];

    // Preload next segment while current is playing
    if (i + 1 < segments.length) {
      getAudioUrl(segments[i + 1].text, segments[i + 1].style); // Fire and forget
    }

    const url = await getAudioUrl(text, style);
    if (!url) continue; // Silent skip if unavailable

    await new Promise((resolve) => {
      const audio = new Audio(url);
      audio.onended = resolve;
      audio.onerror = resolve; // Never block on error
      audio.play().catch(resolve);
    });
  }
}
