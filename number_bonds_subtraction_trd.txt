1. TECHNICAL OVERVIEW

This document specifies the architecture, component design, state management, data models, simulation logic, gamification implementation, and quality standards for the "Number Bonds for Subtraction" interactive lesson module (Lesson 3.2) within Intellia SG's Grade 1 Math program.

The module is a React 18 application (Vite + JSX), structured identically to the reference repository https://github.com/dsamyak/readingnumbers, and styled to match https://readingnumbers.vercel.app/. It will be embedded in WordPress at:
https://intelliasg.com/courses/grade-1-math/lessons/number-bonds-for-subtraction/

5. QUESTION DATA MODEL

5.1 Question Schema (TypeScript-style)

interface Question {
  id: string;                  // e.g. "Q1_007", "Q4_015"
  type: QuestionType;          // Enum (10 types — see §7.1 of PRD)
  world: number;               // 0–9 (which world it belongs to)
  difficulty: 1 | 2 | 3;      // 1=easy(≤10), 2=medium(≤15), 3=hard(≤20)
  whole: number;               // The minuend (total / whole in bond)
  part1: number;               // Known subtrahend
  part2: number;               // Difference (the answer)
  missingSlot: 'whole' | 'part1' | 'part2'; // What is hidden

  // Rendering
  questionText: string;        // Full question text
  visual: 'bond' | 'counters' | 'picture' | 'sentence' | 'trueFalse';
  counterEmoji?: string;       // '🍎' | '⭐' | '🍪' | '🎈' etc.
  options?: (string|number)[]; // MCQ: always 4 options

  // Hints
  hint1: string;               // Highlight bond diagram
  hint2: string;               // Show counter animation
  explanation: string;         // Full explanation after 2 fails

  // Singapore context (word problems only)
  characterName?: string;      // 'Wei Ming', 'Priya' etc.
  objectName?: string;         // 'stickers', 'cookies' etc.

  // Answer
  correctAnswer: number | string;
}

type QuestionType =
  | 'bond_diagram'          // Q1: Number bond with missing part
  | 'fill_blank'            // Q2: ___ – ___ = ___
  | 'picture_takeaway'      // Q3: Emoji counters, some crossed
  | 'word_problem'          // Q4: Singapore narrative
  | 'fact_family'           // Q5: Complete the bond family
  | 'true_false'            // Q6: Is this bond correct?
  | 'match_bond'            // Q7: Which bond matches the sentence?
  | 'number_before_after'   // Q8: Find the whole from parts
  | 'within_20'             // Q9: Double-digit subtraction
  | 'find_whole';           // Q10: Reverse — what is the whole?

8. RANDOMISATION ENGINE

8.1 Fisher-Yates Shuffle

// utils/shuffle.js
export function shuffleArray(array) {
  const arr = [...array];
  for (let i = arr.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [arr[i], arr[j]] = [arr[j], arr[i]];
  }
  return arr;
}

export function generateSessionQuestions(bank) {
  // Ensure distribution: pick 10 from each type, then shuffle
  const byType = {};
  bank.forEach(q => {
    if (!byType[q.type]) byType[q.type] = [];
    byType[q.type].push(q);
  });
  const selected = Object.values(byType).flatMap(qs => shuffleArray(qs).slice(0, 10));
  return shuffleArray(selected); // Final shuffle of all 100
}

8.2 MCQ Distractor Generation

// utils/scoring.js
export function generateDistractors(correct, min = 0, max = 20, count = 3) {
  const distractors = new Set();
  let attempts = 0;
  while (distractors.size < count && attempts < 50) {
    const offset = (Math.random() > 0.5 ? 1 : -1) * (Math.ceil(Math.random() * 3));
    const d = correct + offset;
    if (d >= min && d <= max && d !== correct) distractors.add(d);
    attempts++;
  }
  // Fallback distractors if not enough found
  [correct-1, correct+1, correct+2].forEach(d => {
    if (d >= min && d <= max && d !== correct && distractors.size < count)
      distractors.add(d);
  });
  return shuffleArray([correct, ...distractors]);
}

9.1 XP Calculation

// utils/scoring.js
const XP_TABLE = {
  correctFirstTry:  10,
  correctSecondTry:  7,
  correctWithHint:   5,
};

export function calcXP(attemptNumber, hintsUsed, streak) {
  const base = attemptNumber === 1 ? 10 : hintsUsed > 0 ? 5 : 7;
  const streakBonus = streak >= 5 ? 5 : 0;
  return base + streakBonus;
}

9.2 Star Rating (per world / per 10 questions)

export function calcStars(correct, total = 10) {
  if (correct >= 9) return 3;
  if (correct >= 7) return 2;
  if (correct >= 5) return 1;
  return 0;
}

// World unlock gate: must have ≥ 1 star (≥ 5/10 correct)
export function canUnlockWorld(worldScore) {
  return worldScore !== null && worldScore >= 5;
}

10. AUDIO IMPLEMENTATION

10.1 Web Speech API Hook

// hooks/useAudio.js
export function useAudio(audioEnabled) {
  const speak = (text, options = {}) => {
    if (!audioEnabled || !window.speechSynthesis) return;
    window.speechSynthesis.cancel();
    const utterance = new SpeechSynthesisUtterance(text);
    utterance.rate  = options.rate  ?? 0.85;   // Slower for children
    utterance.pitch = options.pitch ?? 1.1;    // Slightly higher pitch
    utterance.lang  = options.lang  ?? 'en-SG';// Singapore English
    window.speechSynthesis.speak(utterance);
  };

  const speakQuestion = (question) => speak(question.questionText);
  const speakNumber   = (n) => speak(String(n));
  const speakPraise   = () => speak(['Amazing!', 'Well done!', 'Brilliant!',
                                     'You got it!'][Math.floor(Math.random() * 4)]);
  const speakHint     = (hintText) => speak(hintText, { rate: 0.8 });

  return { speak, speakQuestion, speakNumber, speakPraise, speakHint };
}

10.2 Sound Effects (Web Audio API)

// hooks/useAudio.js (continued)
const playTone = (frequencies, durations) => {
  const ctx = new (window.AudioContext || window.webkitAudioContext)();
  frequencies.forEach((freq, i) => {
    const osc = ctx.createOscillator();
    const gain = ctx.createGain();
    osc.connect(gain);
    gain.connect(ctx.destination);
    osc.frequency.value = freq;
    gain.gain.setValueAtTime(0.3, ctx.currentTime + (durations[i - 1] || 0) / 1000);
    gain.gain.exponentialRampToValueAtTime(0.001, ctx.currentTime + (durations[i] || 200) / 1000 + 0.5);
    osc.start(ctx.currentTime + (durations[i - 1] || 0) / 1000);
    osc.stop(ctx.currentTime + (durations[i] || 200) / 1000 + 0.5);
  });
};
